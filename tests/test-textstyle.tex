% From http://tex.stackexchange.com/a/274456

\newenvironment{talign}{%
  \let\displaystyle\textstyle\align
}{%
  \endalign
}

\newenvironment{salign}{%
  \let\displaystyle\scriptstyle\align
}{%
  \endalign
}

\newenvironment{ssalign}{%
  \let\displaystyle\scriptscriptstyle\align
}{%
  \endalign
}
